                        class Math
                          def initialize
                            @log = ""
                          end

                          def add1(x) do
                            @log += "add1"
                            x + 1
                          end

                          def square(x) do
                            @log += "square"
                            x * x
                          end
                        end

                        m = Math.new
                        m.square(m.add(1)) # => 4
                        m.log # => "add1square"
